1. Create `src/rapid_viewer/ui/main_window.py`:

```python
from pathlib import Path
from PyQt6.QtWidgets import QMainWindow, QFileDialog, QMessageBox
from PyQt6.QtGui import QAction

APP_TITLE = "ABB RAPID Toolpath Viewer"

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle(APP_TITLE)
        self.setMinimumSize(800, 600)
        self._parse_result = None
        self._setup_menu()

    def _setup_menu(self):
        menu_bar = self.menuBar()
        file_menu = menu_bar.addMenu("&File")

        open_action = QAction("&Open...", self)
        open_action.setShortcut("Ctrl+O")
        open_action.triggered.connect(self._open_file)
        file_menu.addAction(open_action)

        file_menu.addSeparator()

        exit_action = QAction("E&xit", self)
        exit_action.setShortcut("Ctrl+Q")
        exit_action.triggered.connect(self.close)
        file_menu.addAction(exit_action)

    def _open_file(self):
        file_path, _ = QFileDialog.getOpenFileName(
            self,
            "Open RAPID Module",
            "",
            "RAPID Module (*.mod);;All Files (*)"
        )
        if file_path:
            self.load_file(file_path)

    def load_file(self, file_path: str):
        """Load and parse a .mod file. Public method for testing."""
        path = Path(file_path)
        try:
            from rapid_viewer.parser.rapid_parser import parse_module, read_mod_file
            source = read_mod_file(path)
            self._parse_result = parse_module(source)
            self.setWindowTitle(f"{APP_TITLE} - {path.name}")
        except Exception as e:
            QMessageBox.critical(self, "Error", f"Failed to load file:\n{e}")

    @property
    def parse_result(self):
        return self._parse_result
```

2. Create `src/rapid_viewer/main.py`:

```python
import sys
from PyQt6.QtWidgets import QApplication
from rapid_viewer.ui.main_window import MainWindow

def main():
    app = QApplication(sys.argv)
    window = MainWindow()
    window.show()

    # If a file path was passed as argument, load it
    if len(sys.argv) > 1:
        window.load_file(sys.argv[1])

    sys.exit(app.exec())

if __name__ == "__main__":
    main()
```

"""Tests for MainWindow file loading and title bar update."""
import pytest
from pathlib import Path

FIXTURES_DIR = Path(__file__).parent / "fixtures"

def test_title_update_after_load(qtbot):
    """FILE-02: Window title updates with filename after loading .mod file."""
    from rapid_viewer.ui.main_window import MainWindow, APP_TITLE
    window = MainWindow()
    qtbot.addWidget(window)

    # Title starts with app name only
    assert window.windowTitle() == APP_TITLE

    # Load a fixture file
    fixture_path = str(FIXTURES_DIR / "simple.mod")
    window.load_file(fixture_path)

    # Title now includes filename
    assert "simple.mod" in window.windowTitle()
    assert APP_TITLE in window.windowTitle()

def test_parse_result_stored_after_load(qtbot):
    """Verify parse result is accessible after file load."""
    from rapid_viewer.ui.main_window import MainWindow
    window = MainWindow()
    qtbot.addWidget(window)

    assert window.parse_result is None

    fixture_path = str(FIXTURES_DIR / "simple.mod")
    window.load_file(fixture_path)

    assert window.parse_result is not None
    assert len(window.parse_result.moves) > 0

def test_initial_window_size(qtbot):
    """Window has reasonable minimum size."""
    from rapid_viewer.ui.main_window import MainWindow
    window = MainWindow()
    qtbot.addWidget(window)

    assert window.minimumWidth() >= 800
    assert window.minimumHeight() >= 600
