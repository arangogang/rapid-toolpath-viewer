```python
"""Arcball camera for interactive 3D toolpath viewing.

Implements orbit (left-drag), pan (middle-drag), and zoom (scroll wheel)
using quaternion accumulation to avoid gimbal lock.

Usage in QOpenGLWidget mouse events:
    # mousePressEvent:
    camera.orbit_start(x, y, widget_w, widget_h)
    camera.pan_start(x, y)

    # mouseMoveEvent:
    camera.orbit_update(x, y, widget_w, widget_h)
    camera.pan_update(x, y, widget_w, widget_h)

    # wheelEvent:
    camera.zoom(delta)   # delta = angleDelta().y() / 120.0

    # resizeGL:
    camera.set_aspect(w / max(h, 1))

    # paintGL:
    mvp = camera.mvp()
    glUniformMatrix4fv(loc, 1, GL_FALSE, mvp.flatten())
"""

from __future__ import annotations

import numpy as np
import pyrr


class ArcballCamera:
    """Arcball-style interactive camera.

    All coordinates in mm (RAPID coordinate system).
    Default position: 1000mm from origin, looking toward origin, Y=up.
    """

    def __init__(self) -> None:
        self._rotation: pyrr.Quaternion = pyrr.Quaternion()   # identity
        self._distance: float = 1000.0                         # mm from target
        self._pan_offset: np.ndarray = np.zeros(3, dtype=np.float32)
        self._fov: float = 45.0
        self._aspect: float = 1.0
        self._near: float = 1.0
        self._far: float = 100_000.0  # RAPID coords in mm; paths span meters

        # State for drag operations
        self._orbit_start_vec: np.ndarray = np.zeros(3)
        self._orbit_start_rot: pyrr.Quaternion = pyrr.Quaternion()
        self._pan_start_mouse: np.ndarray = np.zeros(2, dtype=np.float32)
        self._pan_start_offset: np.ndarray = np.zeros(3, dtype=np.float32)
        self._mouse_mode: str | None = None

    # ------------------------------------------------------------------
    # Orbit (left-drag)
    # ------------------------------------------------------------------

    def _screen_to_sphere(self, x: float, y: float, w: float, h: float) -> np.ndarray:
        """Map pixel position to point on unit arcball sphere."""
        sx = (2.0 * x - w) / w
        sy = (h - 2.0 * y) / h
        d2 = sx * sx + sy * sy
        if d2 <= 1.0:
            sz = np.sqrt(1.0 - d2)
        else:
            n = np.sqrt(d2)
            sx, sy, sz = sx / n, sy / n, 0.0
        return np.array([sx, sy, sz], dtype=np.float64)

    def orbit_start(self, x: float, y: float, w: float, h: float) -> None:
        """Record drag start position for orbit."""
        self._orbit_start_vec = self._screen_to_sphere(x, y, w, h)
        self._orbit_start_rot = pyrr.Quaternion(self._rotation)

    def orbit_update(self, x: float, y: float, w: float, h: float) -> None:
        """Update rotation from current drag position."""
        v1 = self._orbit_start_vec
        v2 = self._screen_to_sphere(x, y, w, h)
        dot = float(np.clip(np.dot(v1, v2), -1.0, 1.0))
        axis = np.cross(v1, v2)
        axis_len = np.linalg.norm(axis)
        if axis_len < 1e-6:
            return
        axis = axis / axis_len
        angle = np.arccos(dot)
        delta = pyrr.Quaternion.from_axis(axis, angle * 2.0)
        self._rotation = pyrr.Quaternion(
            (delta * self._orbit_start_rot).normalized
        )

    # ------------------------------------------------------------------
    # Pan (middle-drag)
    # ------------------------------------------------------------------

    def pan_start(self, x: float, y: float) -> None:
        """Record drag start position for pan."""
        self._pan_start_mouse = np.array([x, y], dtype=np.float32)
        self._pan_start_offset = self._pan_offset.copy()

    def pan_update(self, x: float, y: float, w: float, h: float) -> None:
        """Update pan offset from current drag position."""
        dx = (x - self._pan_start_mouse[0]) / w
        dy = (y - self._pan_start_mouse[1]) / h
        pan_speed = self._distance * 2.0
        rot_mat = pyrr.Matrix44.create_from_quaternion(self._rotation)
        right = np.asarray(rot_mat)[0, :3]
        up = np.asarray(rot_mat)[1, :3]
        self._pan_offset = (
            self._pan_start_offset
            - right * dx * pan_speed
            + up * dy * pan_speed
        ).astype(np.float32)

    # ------------------------------------------------------------------
    # Zoom (scroll wheel)
    # ------------------------------------------------------------------

    def zoom(self, delta: float) -> None:
        """Adjust camera distance. delta > 0 = zoom in, delta < 0 = zoom out."""
        factor = 1.0 - delta * 0.1
        self._distance = max(10.0, self._distance * factor)

    # ------------------------------------------------------------------
    # Camera configuration
    # ------------------------------------------------------------------

    def set_aspect(self, aspect: float) -> None:
        """Update viewport aspect ratio (width / height)."""
        self._aspect = float(aspect)

    def reset(self, center: np.ndarray, scene_radius: float) -> None:
        """Fit camera to scene bounding sphere. Called after file load."""
        self._rotation = pyrr.Quaternion()
        self._pan_offset = center.astype(np.float32)
        self._distance = max(scene_radius * 2.5, 100.0)

    @property
    def rotation(self) -> pyrr.Quaternion:
        return self._rotation

    # ------------------------------------------------------------------
    # Matrix generation
    # ------------------------------------------------------------------

    def view_matrix(self) -> np.ndarray:
        """4x4 float32 view matrix."""
        rot_mat = pyrr.Matrix44.create_from_quaternion(self._rotation)
        rot_arr = np.asarray(rot_mat, dtype=np.float64)
        eye_local = np.array([0.0, 0.0, self._distance, 1.0])
        eye = (rot_arr @ eye_local)[:3] + self._pan_offset
        target = self._pan_offset
        up = (rot_arr @ np.array([0.0, 1.0, 0.0, 0.0]))[:3]
        return pyrr.matrix44.create_look_at(eye, target, up, dtype=np.float32)

    def projection_matrix(self) -> np.ndarray:
        """4x4 float32 perspective projection matrix."""
        return pyrr.matrix44.create_perspective_projection(
            self._fov, self._aspect, self._near, self._far, dtype=np.float32
        )

    def mvp(self) -> np.ndarray:
        """Full MVP matrix (model = identity for toolpath scene)."""
        return (self.projection_matrix() @ self.view_matrix()).astype(np.float32)
```
